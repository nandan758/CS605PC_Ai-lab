from IPython.display import display, HTML class VacuumEnvironment: def __init__(self, rooms, start_location): self.rooms = rooms self.location = start_location def percept(self): return (self.location, self.rooms[self.location]) def execute(self, action): if action == 'Suck': self.rooms[self.location] = 'Clean' elif action == 'Right': self.location = 'B' elif action == 'Left': self.location = 'A' def render(self, action=None): html = f"""

Vacuum World

Room Status
A {self.rooms['A']}
B {self.rooms['B']}

Vacuum Location: {self.location}

Last Action: {action}

""" display(HTML(html))
